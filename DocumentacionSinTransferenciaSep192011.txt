        function ValidarMod(){

                    j++ ;
            }
            if ( j = 0 ){
                alert("Para modificar tiene que seleccionar un recurso")
            }else if ( j > 1 ){
                alert("Solo puede seleccionar un campo")
            }else{
                document.action = 'LA OTRA ACCION'

            }
        }
